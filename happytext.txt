CREATE TABLE clients (
    id             INTEGER NOT NULL
                           PRIMARY KEY AUTOINCREMENT,
    surname        TEXT    NOT NULL,
    name           TEXT    NOT NULL,
    [phone number] CHAR    NOT NULL
                           UNIQUE,
    registration   DATE    NOT NULL
);
CREATE TABLE goods (
    id          INTEGER PRIMARY KEY AUTOINCREMENT
                        NOT NULL,
    [good name] TEXT    NOT NULL
                        UNIQUE,
    price       INTEGER NOT NULL,
    avaliable   BOOLEAN NOT NULL,
    description TEXT    NOT NULL
);
CREATE TABLE orders (
    id            INTEGER PRIMARY KEY AUTOINCREMENT
                          NOT NULL,
    client_id     INTEGER REFERENCES clients (id) 
                          NOT NULL,
    data          DATE    NOT NULL,
    [order price] INTEGER NOT NULL,
    good_id       INTEGER REFERENCES goods (id) 
                          NOT NULL
);

UPDATE clients
 SET registration = SUBSTR(registration,7,4) || '-' || SUBSTR(registration,4,2) || '-' || SUBSTR(registration,1,2);
SELECT * from clients;
